JavaScript: 动态创建命名空间

 

let App = {};
App.namespace = function(name) {
  let parts = name.split('.');
  let current = App;
  for (let i in parts) {
    if (!current[parts[i]]) { current[parts[i]] = {}; }
    current = current[parts[i]];
  }
};
​
App.namespace('event');
App.namespace('configMap.html');

上面的代码等价于


let App = {
  event: {},
  configMap: {
    html: {}
  }
};
