JavaScript: 外部迭代器

let Iterator = function(obj) {
  let current = 0;
  let next = function() { current += 1; };
  let isDone = function() { return current >= obj.length; };
  let getCurrItem = function() { return obj[current]; };
  let len = function() { return obj.length; };
​
  return {
    next: next,
    isDone: isDone,
    getCurrItem: getCurrItem,
    len: len
  };
};
​
let compare = function(iterator1, iterator2) {
  while (!iterator1.isDone() && !iterator2.isDone()) {
    if (iterator1.len() !== iterator2.len()) {
      throw new Error('iterator1 和 iterator2 不相等');
    }
    if (iterator1.getCurrItem() !== iterator2.getCurrItem()) {
      throw new Error('iterator1 和 iterator2 不相等');
    }
    iterator1.next();
    iterator2.next();
  }
  console.log('iterator1 和 iterator2 相等');
};
​
let iterator1 = Iterator([1,2,3]);
let iterator2 = Iterator([1,2,3,4]);
​
compare(iterator1, iterator2);
