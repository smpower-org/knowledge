gulp 环境下 JavaScript 代码压缩工具

  1. 安装 gulp-uglify
    $ npm install --save-dev gulp-uglify
  2. 配置 gulp-uglify
    const
      gulp = require('gulp'),
      uglify = require('gulp-uglify'),
    ​
      config = {
        watch: {
          js: 'src/js/**/*.js'  // 监听 src/js 文件夹下所有 js 文件
        },
        build: {
          js: 'web/js'  // 输出压缩后的 js 文件
        },
        tasks: {
          uglify: function() {
            gulp
              .src(config.watch.js)
              .pipe(uglify())
              .pipe(gulp.dest(config.build.js));
          }
        }
      },
      
      gulp.task('uglify', config.tasks.uglify),
      
      gulp.watch('watch', function() {
        gulp.watch(config.watch.js, ['uglify']);
      }),
       
      gulp.task('default', ['watch'])
    ;
