gulp 环境下 HTML 代码压缩工具

  1. 安装 gulp-minify-html 

    $ npm install --save-dev gulp-minify-html
  2. 配置 gulp-minify-html
    const
      gulp = require('gulp'),
      minifyHtml = require('gulp-minify-html'),
    ​
      config = {
        watch: {
          html: 'src/**/*.html'  // 监听 src 文件夹下所有 html 文件
        },
        build: {
          html: 'web/html'  // 输出压缩后的 html 文件
        },
        tasks: {
          minifyHtml: function() {
            gulp
              .src(config.watch.html)
              .pipe(minifyHtml())
              .pipe(gulp.dest(config.build.html));
          }
        }
      },
      
      gulp.task('minifyHtml', config.tasks.minifyHtml),
      
      gulp.watch('watch', function() {
        gulp.watch(config.watch.html, ['minifyHtml']);
      }),
       
      gulp.task('default', ['watch'])
    ;
