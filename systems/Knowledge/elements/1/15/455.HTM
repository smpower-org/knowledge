gulp 环境下 CSS 代码压缩工具

  1. 安装 gulp-minify-css 
    $ npm install --save-dev gulp-minify-css
  2. 配置 gulp-minify-css
    const
      gulp = require('gulp'),
      minifyCss = require('gulp-minify-css'),
    ​
      config = {
        watch: {
          css: 'src/css/**/*.css'  // 监听 src/css 文件夹下所有 css 文件
        },
        build: {
          js: 'web/css'  // 输出压缩后的 css 文件
        },
        tasks: {
          minifyCss: function() {
            gulp
              .src(config.watch.css)
              .pipe(minifyCss())
              .pipe(gulp.dest(config.build.css));
          }
        }
      },
      
      gulp.task('minifyCss', config.tasks.minifyCss),
      
      gulp.watch('watch', function() {
        gulp.watch(config.watch.css, ['minifyCss']);
      }),
       
      gulp.task('default', ['watch'])
