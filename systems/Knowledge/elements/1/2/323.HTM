给输入框添加多个校验规则

HTML: 

<div class="register">
  <input class="username" type="text" placeholder="请输入用户名">
  <input class="password" type="text" placeholder="请输入密码">
  <input class="phone" type="text" placeholder="请输入手机号码">
  <button class="submit">Submit</button>
</div>

JavaScript: 

/****** 策略对象 ******/
let strategies = {
  isNonEmpty: function(value, errorMsg) {  // 校验输入框是否为空
    if (value === '') { return errorMsg; }
  },
  minLength: function(value, length, errorMsg) {  // 校验输入框中的值的长度
    if (value.length < length) { return errorMsg; }
  },
  maxLength: function(value, length, errorMsg) {  // 校验输入框中的值的长度
    if (value.length > length) { return errorMsg; }
  },
  isMobile: function(value, errorMsg) {  // 校验手机号码格式
    if (!/^1[3|5|8][0-9]{9}$/.test(value)) { return errorMsg; }
  }
};
​
​
/****** Validator 类 ******/
let Validator = function() {
  this.cache = [];  // 缓存校验规则
};
​
Validator.prototype.add = function(dom, rules) {
  let self = this;
​
  for (let i = 0, len = rules.length; i < len; i++) {
    (function(rule) {
      let strategyArr = rule.strategy.split(':');
      let errorMsg = rule.errorMsg;
​
      self.cache.push(function() {
        let strategy = strategyArr.shift();
        strategyArr.unshift(dom.value);
        strategyArr.push(errorMsg);
        return strategies[strategy].apply(dom, strategyArr);
      });
    }(rules[i]));
  }
};
​
Validator.prototype.start = function() {
  // 遍历缓存校验规则的数组 cache，依次执行其中的方法
  for (let i = 0, len = this.cache.length; i < len; i++) {
    let validate = this.cache[i];
    let errorMsg = validate();
    if (errorMsg) { return errorMsg; }
  }
};
​
​
/****** 客户调用代码 ******/
let register = document.querySelector('.register');
let username = register.querySelector('.username');
let inputs = register.querySelectorAll('input');
​
let validatorFunc = function() {
  let validator = new Validator();  // 初始化一个新的 validator 验证对象
​
  // 对 username 添加多种校验规则
  validator.add(username, [{
    strategy: 'isNonEmpty',
    errorMsg: '用户名不能为空'
  }, {
    strategy: 'minLength: 8',
    errorMsg: '用户名长度不能小于 8 位'
  }, {
    strategy: 'maxLength: 12',
    errorMsg: '用户名长度不能大于 12 位'
  }]);
​
  let errorMsg = validator.start();  // 如果有返回值，说明验证未通过
  return errorMsg;  // 返回错误消息
};
​
for (let i = 0, len = inputs.length; i < len; i++) {
  (function(input) {
    input.onkeyup = function() {
      let errorMsg = validatorFunc();
      if (errorMsg) {
        console.log(errorMsg);
        return false;
      }
    };
  }(inputs[i]));
}
