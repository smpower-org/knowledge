JavaScript: 虚拟代理实现图片预加载

HTML: 

<div class="myImage"></div>

CSS: 

.myImage {
  width: 100%;
  display: flex;
  justify-content: center;
  align-items: center;
}
.myImage img {
  width: 100%;
}

JavaScript: 

let myImage = (function() {
  let imgNode = document.createElement('img');
  document.querySelector('.myImage').appendChild(imgNode);
​
  return {
    setSrc: function(src) {
      imgNode.src = src;
    }
  };
}());
